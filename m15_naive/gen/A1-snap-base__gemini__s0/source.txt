from build123d import *

# Design Parameters
width = 60.0
length = 80.0
height = 40.0
wall = 2.0
rad = 5.0

# --- 1. Create the Snap Ridge Sketches ---
ridge_length = length - 2 * rad - 10
ridge_z = height - 6
ridge_h = 1.6
ridge_proj = 0.5
ridge_top_h = 0.4

# Right ridge profile on XZ plane
with BuildSketch(Plane.XZ) as right_ridge_sketch:
    p1 = (width / 2, ridge_z - ridge_h / 2)
    p2 = (width / 2 + ridge_proj, ridge_z - ridge_top_h / 2)
    p3 = (width / 2 + ridge_proj, ridge_z + ridge_top_h / 2)
    p4 = (width / 2, ridge_z + ridge_h / 2)
    with BuildLine():
        Polyline([p1, p2, p3, p4, p1])
    make_face()

# Left ridge profile on XZ plane
with BuildSketch(Plane.XZ) as left_ridge_sketch:
    p1 = (-width / 2, ridge_z - ridge_h / 2)
    p2 = (-width / 2 - ridge_proj, ridge_z - ridge_top_h / 2)
    p3 = (-width / 2 - ridge_proj, ridge_z + ridge_top_h / 2)
    p4 = (-width / 2, ridge_z + ridge_h / 2)
    with BuildLine():
        Polyline([p1, p2, p3, p4, p1])
    make_face()

# --- 2. Create the Box ---
with BuildPart() as box_part:
    with BuildSketch() as s1:
        RectangleRounded(width, length, rad)
    extrude(amount=height)
    
    # Shell the box
    top_face = box_part.faces().sort_by(Axis.Z)[-1]
    offset(amount=-wall, openings=[top_face])
    
    # Add the ridges
    extrude(right_ridge_sketch.sketch, amount=ridge_length, dir=(0, 1, 0), both=True)
    extrude(left_ridge_sketch.sketch, amount=ridge_length, dir=(0, 1, 0), both=True)

# --- 3. Create the Lid ---
lid_clearance = 0.25
lid_wall = 2.0
lid_inner_height = 10.0
lid_inner_width = width + 2 * lid_clearance
lid_inner_length = length + 2 * lid_clearance
lid_inner_rad = rad + lid_clearance
lid_outer_width = lid_inner_width + 2 * lid_wall
lid_outer_length = lid_inner_length + 2 * lid_wall
lid_outer_rad = lid_inner_rad + lid_wall
lid_outer_height = lid_inner_height + lid_wall

# Groove parameters
groove_length = ridge_length + 1.0
groove_z = lid_outer_height - 6
groove_h = 1.8
groove_depth = 0.5
groove_top_h = 0.6

# Right groove profile on XZ plane
with BuildSketch(Plane.XZ) as right_groove_sketch:
    p1 = (lid_inner_width / 2, groove_z - groove_h / 2)
    p2 = (lid_inner_width / 2 + groove_depth, groove_z - groove_top_h / 2)
    p3 = (lid_inner_width / 2 + groove_depth, groove_z + groove_top_h / 2)
    p4 = (lid_inner_width / 2, groove_z + groove_h / 2)
    with BuildLine():
        Polyline([p1, p2, p3, p4, p1])
    make_face()

# Left groove profile on XZ plane
with BuildSketch(Plane.XZ) as left_groove_sketch:
    p1 = (-lid_inner_width / 2, groove_z - groove_h / 2)
    p2 = (-lid_inner_width / 2 - groove_depth, groove_z - groove_top_h / 2)
    p3 = (-lid_inner_width / 2 - groove_depth, groove_z + groove_top_h / 2)
    p4 = (-lid_inner_width / 2, groove_z + groove_h / 2)
    with BuildLine():
        Polyline([p1, p2, p3, p4, p1])
    make_face()

with BuildPart() as lid_part:
    with BuildSketch() as s2:
        RectangleRounded(lid_outer_width, lid_outer_length, lid_outer_rad)
        # Add pull tabs on the short sides
        with Locations((0, lid_outer_length / 2), (0, -lid_outer_length / 2)):
            RectangleRounded(15, 8, 2)
    extrude(amount=lid_outer_height)
    
    # Shell the lid
    bottom_face = lid_part.faces().sort_by(Axis.Z)[0]
    offset(amount=-lid_wall, openings=[bottom_face])
    
    # Subtract the grooves
    extrude(right_groove_sketch.sketch, amount=groove_length, dir=(0, 1, 0), both=True, mode=Mode.SUBTRACT)
    extrude(left_groove_sketch.sketch, amount=groove_length, dir=(0, 1, 0), both=True, mode=Mode.SUBTRACT)

# Translate the lid so they are side-by-side for printing/viewing
box_solid = box_part.part
lid_solid = lid_part.part.translate((0, length + 20, 0))

result = [box_solid, lid_solid]
